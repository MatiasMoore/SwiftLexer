func dfs(_ array: [[Int]], _ row: Int, _ col: Int, _ rows: Int, _ cols: Int, _ startColor: Int, _ newColor: Int) -> [[Int]] {
    // Создаём копию массива для работы
    var newArray = array

    // Проверка границ массива
    if row < 0 || row >= rows || col < 0 || col >= cols {
        return newArray // Выход из рекурсии при выходе за границы
    }

    // Проверка цвета
    if newArray[row][col] != startColor {
        return newArray // Выход из рекурсии, если цвет не совпадает
    }

    // Замена цвета
    newArray[row][col] = newColor

    // Рекурсивные вызовы, передавая новый массив
    var result = dfs(newArray, row + 1, col, rows, cols, startColor, newColor) // вниз
    result = dfs(result, row - 1, col, rows, cols, startColor, newColor) // вверх
    result = dfs(result, row, col + 1, rows, cols, startColor, newColor) // вправо
    result = dfs(result, row, col - 1, rows, cols, startColor, newColor) // влево
    return result // Возвращаем обновленный массив после всех рекурсивных вызовов
}

func floodFill(_ array: [[Int]], _ x: Int, _ y: Int, _ newColor: Int) -> [[Int]] {
    // Получаем размеры массива динамически
    let rows = array.count
    let cols = array[0].count

    // Проверка на корректность начальных координат
    if x < 0 || x >= rows || y < 0 || y >= cols {
        print("Начальные координаты за пределами массива")
        return array
    }

    let startColor = array[x][y]

    // Если цвет совпадает, возвращаем оригинальный массив
    if startColor == newColor {
        print("Цвета совпадают, заливка не требуется")
        return array
    }

    // Запускаем DFS
    return dfs(array, x, y, rows, cols, startColor, newColor) // Исправлено: передаем rows и cols
}
