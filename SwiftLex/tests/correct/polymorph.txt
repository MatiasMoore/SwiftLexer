class Parent{
	public var a = 199

	public func parentFuncToOverride() {
		print("Parent Func to override!")
	}

}

class Child : Parent{
	override public func parentFuncToOverride() {
		print("Chindren overriden Func!")
	}
}


var arr : [Parent] = [Parent(), Child(), Parent(), Child()]

for elem in arr
{
	elem.parentFuncToOverride();
}
