class Parent{
	public var a = 199

	public func parentFuncToOverride() {
		print("Parent Func to override!")
	}

}

class Child : Parent{
	override public func parentFuncToOverride() {
		print("Child 1 overriden Func!")
	}
}

class Child2 : Parent{

	override public func parentFuncToOverride() {
		super.parentFuncToOverride()
		print("Child 2 overriden Func!")
	}
}


var arr : [Parent] = [Parent(), Child(), Child2()]

for elem in arr
{
	print("New elem:")
	elem.parentFuncToOverride();
}
