class Test2
{
	public static var testVariable : Int; 
	static public func testFunc()
	{
		print(2)
	}
}

class Test
{
	static public func testFunc(i : Int)
	{
		print(i)
	}

	static public func changeVariable(i : Int)
	{
		Test2.testVariable = i;
	}


}
Test2.testVariable = 99;
Test.testFunc(Test2.testVariable)
Test.changeVariable(123456789)
Test.testFunc(Test2.testVariable)
