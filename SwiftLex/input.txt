class A {
	public var a : Int
	init()
	{
	self.a = 32
	print(self.a)
	}

	public func printField()
	{
		print(self.a)
	}
}

class B : A {
	public var b : A
	init()
	{
	self.a = 2323
	self.b = A()
	self.b.a = 999
	print(self.a)
	}

	public func printBField()
	{
		print(self.a)
		print(self.b.a)
	}
}


var a = A()
a.a = 10
a.printField()

var b = B()
b.a = 123
b.b.a = 1999
b.printBField()
