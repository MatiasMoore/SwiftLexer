	public func printNum()
	{
		print(5)
	}
}

class A {
	public func printNum()
	{
		print(5)
	}
}

class B : A
{
	public override func printNum()
	{
		print(6)
	}
}

public static func polyFunc(aObj : A)
{
	aObj.printNum()
}

var bObj = B()

polyFunc(bObj)

var bObjAsA : A = bObj

var cObj = C()
